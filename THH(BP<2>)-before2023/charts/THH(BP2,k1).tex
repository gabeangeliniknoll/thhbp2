\NewSseqGroup\truncatedvonetower{m}{
	\class(0,0)
	\foreach \n in {1, ..., #1}{
		\class(2*\n, \n)
		\structline(2*(\n-1), \n-1)(2*\n, \n)
	}
}

\DeclareSseqGroup\infvonetower{}{
	\class(0,0)
	\DoUntilOutOfBoundsThenNMore{3}{
	\class(\lastx + 2, \lasty+1)
	\structline
	}
}

\DeclareSseqGroup \tower {} {
    \class(0,0)
    \DoUntilOutOfBounds {
        \class(\lastx,\lasty+1)
        \structline
	} 
}
